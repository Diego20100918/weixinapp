        function exportToPDF() {

            const element = document.getElementById('chat-window');

            const dateStr = new Date().toLocaleDateString().replace(/\//g, '-');

            const fileName = `${currentSubject}_學習筆記_${dateStr}.pdf`;


            // 配置導出選項

            const opt = {

                margin:       10,

                filename:     fileName,

                image:        { type: 'jpeg', quality: 0.98 },

                html2canvas:  { 

                    scale: 2, 

                    useCORS: true,

                    logging: false 

                },

                jsPDF:        { unit: 'mm', format: 'a4', orientation: 'portrait' }

            };


            // 執行導出

            appendSystemMessage("正在生成 PDF 檔案，請稍候...");

            html2pdf().set(opt).from(element).save().then(() => {

                console.log("PDF Exported successfully");

            });

        }


        function setSubject(sub) {

            currentSubject = sub;

            document.getElementById('sidebar-subject').innerText = sub;

            const buttons = document.querySelectorAll('.sub-btn');

            buttons.forEach(btn => {

                if(btn.innerText === sub) {

                    btn.className = "sub-btn px-4 py-1.5 rounded-lg bg-white shadow-sm text-indigo-700 text-xs font-bold";

                } else {

                    btn.className = "sub-btn px-4 py-1.5 rounded-lg text-slate-500 text-xs font-medium hover:bg-white/50";

                }

            });

        }


        async function sendMessage() {

            const input = document.getElementById('user-input');

            const btn = document.getElementById('send-btn');

            const text = input.value.trim();

            const selectedModel = document.getElementById('model-select').value;

            

            if (!text) return;


            appendMessage('user', text);

            input.value = '';

            btn.disabled = true;

            

            chatHistory.push({ role: "user", content: `(科目:${currentSubject}) ${text}` });


            try {

                const config = CONFIG[selectedModel];

                const response = await fetch(config.url, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${config.key}` },

                    body: JSON.stringify({ model: config.model, messages: chatHistory })

                });


                const data = await response.json();

                const aiReply = data.choices[0].message.content;


                appendMessage('ai', aiReply);

                chatHistory.push({ role: "assistant", content: aiReply });


            } catch (error) {

                appendMessage('ai', "連線失敗，請檢查 API。");

            } finally {

                btn.disabled = false;

            }

        }


        function appendMessage(role, text) {

            const chatWindow = document.getElementById('chat-window');

            const msgDiv = document.createElement('div');

            msgDiv.className = `flex items-start gap-4 ${role === 'user' ? 'flex-row-reverse' : ''}`;


            const bubbleClass = role === 'user' ? 'bg-indigo-600 text-white shadow-md' : 'bg-white text-slate-700 border border-slate-100 shadow-sm';


            msgDiv.innerHTML = `

                <div class="w-9 h-9 rounded-xl ${role === 'user' ? 'bg-slate-200 text-slate-600' : 'bg-indigo-600 text-white'} flex items-center justify-center flex-shrink-0 text-xs">

                    <i class="fas ${role === 'user' ? 'fa-user' : 'fa-robot'}"></i>

                </div>

                <div class="${bubbleClass} px-5 py-3 rounded-2xl max-w-[85%] text-sm leading-relaxed">

                    ${text}

                </div>

            `;

            chatWindow.appendChild(msgDiv);

            chatWindow.scrollTop = chatWindow.scrollHeight;


            if (window.MathJax && window.MathJax.typesetPromise) {

                window.MathJax.typesetPromise([msgDiv]);

            }

        }


        function appendSystemMessage(text) {

            const chatWindow = document.getElementById('chat-window');

            const div = document.createElement('div');

            div.className = "flex justify-center my-2";

            div.innerHTML = `<span class="text-[10px] bg-emerald-100 text-emerald-700 px-3 py-1 rounded-full font-bold uppercase tracking-widest">${text}</span>`;

            chatWindow.appendChild(div);

            chatWindow.scrollTop = chatWindow.scrollHeight;

        }
